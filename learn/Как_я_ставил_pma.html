Как я ставил pma

Задача: запустить phpMyAdmin

для работы с БД

1. Качаешь phpMyAdmin c офф сайта;

2. Помещаешь распакованный архив там, где лежат папки с сайтами;
Например, для nginx - это var/www/

3. В файл hosts (etc) - добавляешь новое имя для запуска pmy для локальной машины;
---- 127.0.0.1 phpmyadmin.local ----

4. Добавляешь настройки для запуска pma в папку с nginx (etc/nginx/sites-available);
Например, phpmyadmin.local.conf

server {
  listen 80;
 
  # Multiple hostnames separated by spaces.  Replace these as well.
  server_name phpmyadmin.local; # Alternately: _
 
  root /var/www/phpmyadmin/;
 
  index index.php;
  add_header Access-Control-Allow-Origin *;
 
     # serve static files directly
        location ~* \.(jpg|jpeg|gif|css|png|js|ico|html)$ {
                access_log off;
                expires max;
                log_not_found off;
        }
 
        # removes trailing slashes (prevents SEO duplicate content issues)
     #   if (!-d $request_filename)
     #   {
     #           rewrite ^/(.+)/$ /$1 permanent;
     #   }
 
       
 
   #     # unless the request is for a valid file (image, js, css, etc.), send to bootstrap
   #     if (!-e $request_filename)
   #     {
   #             rewrite ^/(.*)$ /index.php?/$1 last;
   #             break;
   #     }
 
        location / {
        add_header Access-Control-Allow-Origin *;
                try_files $uri $uri/ /index.php?$query_string;
        }
 
        location ~* \.php$ {
        try_files $uri = 404;
        fastcgi_split_path_info ^(.+\.php)(/.+)$;
        fastcgi_pass unix:/var/run/php/php7.0-fpm.sock; #unix:/var/run/php5-fpm.sock; # /var/run/php/php7.0-fpm.sock
        fastcgi_index index.php;
        fastcgi_param SCRIPT_FILENAME $document_root$fastcgi_script_name;
        include fastcgi_params;
    }
 
    location ~ /\.ht {
                deny all;
        }
}


используется: php7.0-fpm.sock
server_name - имя хоста
root - путь до pma

5. Делаешь симлинк
sudo ln -s /etc/nginx/sites-available/ /etc/nginx/sites-enabled/

6. sudo service nginx restart - перезапускаешь nginx
