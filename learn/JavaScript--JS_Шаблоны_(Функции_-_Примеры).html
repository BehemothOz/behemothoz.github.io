Пример функции обратного вызова (callback)

В примере, находим все div элементы и четным меняет цвет фона
Обратить внимание на 9 строчку.
Что хорошо:
Обе функции - универсальны и выполняют одно действие (как и нужно)

var findDiv = function(callback) {
    var allDiv = document.querySelectorAll('.container div'),
        allDivLength,
        arrDiv = [],
        i;

    for ( i = 0, allDivLength = allDiv.length; i < allDivLength; i++ ) {

        if ( typeof callback === 'function' && i % 2 === 0 ) {
            callback(allDiv[i]);
        }

        arrDiv.push(allDiv[i]);
    }

     // return arrDiv;
     console.log(arrDiv);
};

var changeColor =  function(div) {
    div.style.backgroundColor = 'red';
};

findDiv(changeColor);
